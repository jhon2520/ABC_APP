    #Crear Subdataframe spara cada dataframe matemático según el CIIU del usuario

    #Metodo para crear subdataframe
    def crearSubDataFrame(codigo_CIIU, df,nombre_columna):
    sub_df = df[["cuenta",codigo_CIIU]].copy()
    sub_df = sub_df.rename(columns={user_CIIU: nombre_columna})

    return sub_df

    #Creacion de subdataframe según el del usuario
    df_sub_mean = crearSubDataFrame(user_CIIU,df_transpose_grouped_mean,user_CIIU + " mean")
    df_sub_max = crearSubDataFrame(user_CIIU,df_transpose_grouped_max,user_CIIU + " max")
    df_sub_min = crearSubDataFrame(user_CIIU,df_transpose_grouped_min,user_CIIU + " min")
    df_sub_std = crearSubDataFrame(user_CIIU,df_transpose_grouped_std,user_CIIU + " std")


    #UNIR subdataframes en uno sólo

    def mergeDataFrames(dfMean,dfMax,dfMin, dfStd):
    df = pd.merge(dfMean,dfMax, on="cuenta")
    df = pd.merge(df,dfMin, on="cuenta")
    df = pd.merge(df,dfStd, on="cuenta")
    return df


    df_sub_merged = mergeDataFrames(df_sub_mean,df_sub_max,df_sub_min, df_sub_std)

    #Unir subFrames con el dataframe usuario

    df_sub_merged = pd.merge(df_sub_merged,user_dataframe,on="cuenta")


    #Crear indicadores básicos y transformación inicial del df merged

    def crearIndicadores(df):
    
    #promedio - datosUser
    df["diferencia"] = df.iloc[:,1] - df.iloc[:,5]
    #datosUser / promedio
    df["diferencia porcentual"] = ( df.iloc[:,5]/ df.iloc[:,1] )*100
    #promedio + DS
    df["limite superior"] = df.iloc[:,1] + df.iloc[:,4]
    #promedio - ds
    df["limite inferior"] = df.iloc[:,1] - df.iloc[:,4]
    
    #crear condiciones
    # 1 = alerta 0= normal
    df["Alertas"] = np.where(df.iloc[:,5] > df["limite superior"],1, np.where(df.iloc[:,5] < df["limite inferior"], 1,0))
    
    return df
